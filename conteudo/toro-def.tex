\subsection{Toro}
\label{toro-def}

\begin{defi}
     Dado $n\geq 0$, o \textbf{$n$-toro}, denotado por $\mathbb{T}^n$, é definido como o espaço topológico produto $\mathbb{S}^1\times \ldots \times \mathbb{S}^1$ ($n$ fatores).
\end{defi}
